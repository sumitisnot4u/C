
# 說明下列程式的關鍵

```
/* myc_1_1, 我的第一個C程式碼 */ 
#include <stdio.h>
#include <stdlib.h>
int main(void)
{
   printf("Hello 偉大的恩師!\n");   	/* 印出Hello 偉大的恩師! 字串 */
   printf("Hello Taiwan!\n");        /* 印出Hello Taiwan! 字串 */   
   
   system("pause");
   return ;
   printf("Hello World!\n");
}
```
